function downloadCSVFromForm() { const caseBlocks = Array.from(document.querySelectorAll(".case-entry")); const cases = caseBlocks.map(entry => ({ caseId: entry.querySelector(".caseId").value, reasonCode: entry.querySelector(".reasonCode").value, trays: entry.querySelector(".trays").value, tubs: entry.querySelector(".tubs").value })); const header = ["recipient", "phone", "address", "notes", "pickupTask", "worker"]; const row = [ document.getElementById("recipient-search-1").value, document.getElementById("recipient-phone-1").value, document.getElementById("destination-address-1").value, document.getElementById("notes-1").value, document.querySelector('input[name="taskType"]:checked').value, document.getElementById("worker").value ]; const flatCases = cases.flatMap((c, i) => [ `caseId${i+1}`, c.caseId, `reasonCode${i+1}`, c.reasonCode, `trays${i+1}`, c.trays, `tubs${i+1}`, c.tubs ]); const allHeader = header.concat(flatCases.filter((_, i) => i % 2 === 0)); const allRow = row.concat(flatCases.filter((_, i) => i % 2 === 1)); const csvContent = "data:text/csv;charset=utf-8," + [allHeader.join(","), allRow.map(v => `"${v}"`).join(",")].join("\n"); const encodedUri = encodeURI(csvContent); const link = document.createElement("a"); link.setAttribute("href", encodedUri); link.setAttribute("download", "onfleet_task_preview.csv"); document.body.appendChild(link); link.click(); document.body.removeChild(link); }
